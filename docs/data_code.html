Soybean rust prevalence data

Here we load the raw data of Soybean rust prevalence in municipalities across Brazil from 2004/2005 to 2020/2021 crop seasons.

report_all = read.csv("data/1-CAF_Ocorrencias_Jul_30_2021_.csv", encoding = "UTF-8")

Data preparation

In this chunk of data we filter only the reports coming from commercial fields, selects only the columns of interest, put the dates in a standard format, and set the planting dates.

sbr_rep11 = report_all %>%
  dplyr::filter(safra != "2005" & safra != "2006") %>% 
  dplyr::filter(tipodearea == "COMERCIAL") %>%
  dplyr::select(safra, dia, cidade, estadosigla, estadio, latitude, longitude, mesplantio, quinzenaplantio, dataanalise) %>%
  mutate( dia =as.character(dia)) %>% 
  filter(dia != "2001-0001-01") %>%
  mutate(month = month(dia, label = F)) %>% 
  mutate(day = day(dia)) %>%
  separate(safra, sep ="/", remove = F, into = c("ano1", "ano2")) %>% 
  ungroup() %>% 
  mutate(ano = case_when(month > 6 ~ ano1,
                         month <= 5 ~ ano2)) %>% 
  unite(new_date, month, day, sep = "-") %>% 
  unite(dia, ano, new_date, sep = "-") %>% 
  mutate(dia = as.Date(dia)) %>% 
  mutate(plant_date = case_when(quinzenaplantio == 0 ~ "01",
                                quinzenaplantio == 1 ~ "07",
                                quinzenaplantio == 2 ~ "21")) %>%
  mutate(mesplantio = case_when(mesplantio == 0~"10",mesplantio ==1~"11",
                                mesplantio ==2~"12", mesplantio ==3~"12",
                                mesplantio ==8~"8", mesplantio ==9~"9",
                                mesplantio ==10~"10",mesplantio ==11~"11",
                                mesplantio ==12~"12")) %>% 
  dplyr::group_by(safra, cidade, latitude, longitude, estadosigla, plant_date, mesplantio,estadio) %>%
  dplyr::summarize(detection_date = min(dia)) %>%
  # dplyr::filter(estadosigla == "RS"|estadosigla == "PR") %>%
  filter(detection_date != "0001-01-01") %>% 
  mutate(month = month(detection_date, label = TRUE)) %>%
  # mutate(sep_15 = "09-15") %>%
  separate(safra,sep ="/", remove = F, into = c("ano", "exc")) %>% 
  dplyr::select(-exc) %>% 
  unite(ano,mesplantio, col = "start_date", sep = "-") %>%
  unite(start_date,plant_date, col = "start_date", sep = "-") %>% 
  mutate(start_date = ymd(start_date, quiet = T)) %>%
  filter(!is.na(start_date)) %>% 
  mutate(days = start_date - (detection_date)) %>%
  # filter(days<0) %>% 
  mutate(days =  as.numeric(abs(days))) %>%
  ungroup() %>% 
  filter(days<1000)
## `summarise()` regrouping output by 'safra', 'cidade', 'latitude', 'longitude', 'estadosigla', 'plant_date', 'mesplantio' (override with `.groups` argument)

sbr_rep1 = sbr_rep11 %>% 
  mutate(month = as.ordered(case_when(month == "set" ~ "Sep",
                           month == "out" ~ "Oct",
                           month == "nov" ~ "Nov",
                           month == "dez" ~ "Dec",
                           month == "jan" ~ "Jan",
                           month == "fev" ~ "Feb",

  mutate(month = factor(month, levels = c("Sep", "Oct", "Nov", "Dec", "Jan", "Feb", "Mar","Apr","May")) ) %>%

Removing remaining repeated reports per locations

sbr_rep = sbr_rep1 %>% 
  dplyr::group_by(season,start_date, state,location,date_report,Phen_state,month,days) %>%
  dplyr::summarise(lat = mean(lat), lon =  mean(lon)) %>% 
  mutate(Phen_state_int =as.integer(Phen_state)) %>% 
  dplyr::group_by(season,state,location) %>% 
  # group_by(season,start_date, state,location,date_report,month,days) %>% 
  dplyr::filter(Phen_state_int == min(Phen_state_int)) %>% 
  mutate(Phen_state = case_when(Phen_state_int == 1 ~ "VC",
                                Phen_state_int == 2 ~ "V1",
                                Phen_state_int == 3 ~ "V2",
                                Phen_state_int == 4 ~ "V3",
                                Phen_state_int == 5 ~ "Vn",
                                Phen_state_int == 6 ~ "R1",
                                Phen_state_int == 7 ~ "R2",
                                Phen_state_int == 8 ~ "R3",
                                Phen_state_int == 9 ~ "R4",
                                Phen_state_int == 10 ~ "R5",
                                Phen_state_int == 11 ~ "R6",
                                Phen_state_int == 12 ~ "R7",
                                Phen_state_int == 13 ~ "R8",
                                Phen_state_int == 14 ~ "R9")) %>%
  group_by(season,state,location) %>% 
  filter(days == min(days)) %>%
  filter(!duplicated(location)) %>% 
  ungroup() %>% 
  separate(season,sep ="/", remove = F, into = c("Season1", "season")) %>% 
  dplyr::select(-Season1) %>% 
  mutate(sep_15 = "09-15") %>% 
  mutate(year = (as.numeric(season) - 1)) %>% 
  unite(year, sep_15,col = "start_season", sep = "-") %>% 
  mutate(start_season = ymd(start_season)) %>% 
  mutate(days_sep15 = date_report-start_season)%>%
  mutate(week = week(date_report)) %>% 
  mutate(week = case_when(week < 37 ~ week+53,
                          week >= 37 ~ week)) %>% 
  mutate(week = week - 37) %>% 
  mutate(Phen_state = factor(Phen_state, levels =c("VC", "V1", "V2", "V3", "Vn", "R1", "R2", "R3", "R4", "R5", "R6", "R7", "R8", "R9"))) %>% 
  mutate(region = case_when(state %in% north_br ~ "North",
                            state %in% mid_west_br ~"Midwest",
                            state %in% south_br ~"South"),
         region = factor(region, levels = c("South", "Midwest", "North"))) %>% 
  
  
  ## Selecting only these three states for further analysis
  filter(state %in% c("RS","SC","PR"))

  filter(season %in% 2005:2021) %>%
